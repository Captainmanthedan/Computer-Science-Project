CREATE TABLE REFRACTION_TEST
(
    SavedRefractionID INT(5) NOT NULL AUTO_INCREMENT,
    UserID INT(3) NOT NULL,
    SavedRefractionName VARCHAR(60) NOT NULL,
    IncidentAngle FLOAT(5,2) NOT NULL,
    RefractedAngle FLOAT(5,2) NOT NULL,
    RefractiveIndex1 FLOAT(3,2) NOT NULL,
    RefractiveIndex2 FLOAT(3,2) NOT NULL,
    CriticalAngle FLOAT(5,2) NOT NULL,
    DateSaved DATETIME NOT NULL,
    PRIMARY KEY (SavedRefractionID),
    FOREIGN KEY (UserID) REFERENCES LOGIN_TEST_SECURE(UserID),
    UNIQUE (SavedRefractionName)
)
