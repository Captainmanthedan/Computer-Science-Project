CREATE TABLE WAVE_TEST
(
    SavedWaveID INT(5) NOT NULL AUTO_INCREMENT,
    UserID INT(3) NOT NULL,
    SavedWaveName VARCHAR(60) NOT NULL,
    Amplitude FLOAT(5,2) NOT NULL,
    Wavelength FLOAT(5,2) NOT NULL,
    WaveSpeed FLOAT(5,2) NOT NULL,
    Frequency FLOAT(5,2) NOT NULL,
    Period FLOAT(5,2) NOT NULL,
    DateSaved DATETIME NOT NULL,
    PRIMARY KEY (SavedWaveID),
    FOREIGN KEY (UserID) REFERENCES LOGIN_TEST_SECURE(UserID),
    UNIQUE (SavedWaveName)
)
