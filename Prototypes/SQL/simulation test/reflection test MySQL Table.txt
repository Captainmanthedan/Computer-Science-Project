CREATE TABLE REFLECTION_TEST
(
    SavedReflectionID INT(5) NOT NULL AUTO_INCREMENT,
    UserID INT(3) NOT NULL,
    SavedReflectionName VARCHAR(60) NOT NULL,
    Angle FLOAT(5,2) NOT NULL,
    DateSaved DATETIME NOT NULL,
    PRIMARY KEY (SavedReflectionID),
    FOREIGN KEY (UserID) REFERENCES LOGIN_TEST_SECURE(UserID),
    UNIQUE (SavedReflectionName)
)
